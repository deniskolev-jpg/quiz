    .correct { background-color: #c8f7c5; }

    .wrong { background-color: #f7c5c5; }

<body>

  <h1>Harry Potter and the Philosopher's Stone Quiz</h1>

  <div id="quiz"></div>

  <button id="submit">Submit Quiz</button>

  <div id="score"></div>


  <script>

    const quizData = [

      {q:"Who is Harry Potter's best friend?", a:["Hermione","Ron","Draco","Neville"], correct:1, points:5},

      {q:"What house is Harry sorted into?", a:["Slytherin","Ravenclaw","Hufflepuff","Gryffindor"], correct:3, points:5},

      {q:"Who delivers Harry's Hogwarts letter?", a:["Hagrid","Dumbledore","McGonagall","Voldemort"], correct:0, points:5},

      {q:"What position does Harry play in Quidditch?", a:["Seeker","Chaser","Keeper","Beater"], correct:0, points:5},

      {q:"Who is the headmaster of Hogwarts?", a:["Snape","Dumbledore","Hagrid","Filch"], correct:1, points:5},

      {q:"What is the name of Harry's owl?", a:["Crookshanks","Hedwig","Scabbers","Errol"], correct:1, points:5},

      {q:"What creature guards the Philosopher's Stone?", a:["Dragon","Troll","Fluffy","Basilisk"], correct:2, points:5},

      {q:"Who teaches Potions?", a:["Snape","McGonagall","Hagrid","Flitwick"], correct:0, points:5},

      {q:"What does Harry buy first in Diagon Alley?", a:["Wand","Broomstick","Cloak","Cauldron"], correct:0, points:5},

      {q:"Name of Hogwarts train?", a:["Express","Hogwarts Express","Magic Train","Wizard Express"], correct:1, points:5}

      // ... add remaining questions

    ];


    const quizDiv = document.getElementById('quiz');

    const userAnswers = [];


    quizData.forEach((q, idx) => {

      const qDiv = document.createElement('div');

      qDiv.className = 'question';

      qDiv.innerHTML = `<b>Q${idx+1}: ${q.q}</b><br>`;

      q.a.forEach((ans, i) => {

        const btn = document.createElement('button');

        btn.textContent = ans;

        btn.onclick = () => { userAnswers[idx] = i; btn.style.border = "2px solid #000"; };

        qDiv.appendChild(btn);

      });

      quizDiv.appendChild(qDiv);

    });


    document.getElementById('submit').onclick = () => {

      let score = 0;

      quizDiv.querySelectorAll('.question').forEach((qDiv, idx) => {

        const correctIndex = quizData[idx].correct;

        const buttons = qDiv.querySelectorAll('button');

        const chosen = userAnswers[idx];

        if(chosen === correctIndex) score += quizData[idx].points;

        buttons.forEach((b, i) => {

          b.disabled = true;

          if(i === correctIndex) b.classList.add('correct');

          else if(i === chosen && chosen !== correctIndex) b.classList.add('wrong');

        });

      });

      document.getElementById('score').innerText = `Your total score is: ${score}`;

    };

  </script>

</body>
